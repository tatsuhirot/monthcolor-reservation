# 会計（レジ・金種・伝票）内製化 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 予約プレートから起動する会計（明細編集・割引・預かり/お釣り・伝票No・物販）を内製化し、確定で予約レコードに会計情報を書き sales-log に売上を追記する。

**Architecture:** 計算は純粋関数 `lib/checkout.js` に集約。会計確定 `api/checkout.js` は予約（`reservations-queue.json`）を正として `data.checkout` と `data.visitStatus="paid"` を書き、`sales-log.json` に複数明細の売上を追記する。物販は `product-master.json` を `api/products.js` でCRUD。領収書・集計は複数明細＋後方互換で追従。

**Tech Stack:** Node.js (CommonJS), Vercel Serverless 関数形式 `module.exports = async (req,res)=>{}`, Cloudflare R2 (`lib/storage.js`), `node:test` + `node:assert`, 認証 `Authorization: Bearer ${COMINGSOON_PASSWORD}`。

---

## 重要な前提（実装者は必ず読む）

1. **予約レコードの形**（`reservations-queue.json` の1要素）：
   ```json
   { "id": "uuid", "status": "pending|completed|...", "data": {
       "date": "2026-06-17", "time": "12:30", "name": "山田 花子", "nameKana": "",
       "phone": "", "email": "", "menuItems": [...], "menuName": "...", "service": "hair",
       "source": "manual", "nominated": false, "visitStatus": "reserved" },
     "createdAt": "...", "processedAt": null, "error": null }
   ```
   **会計情報・visitStatus は `data` の下にある。** 設計書JSON例はルート表記だが、既存コードに合わせ `data.checkout` / `data.visitStatus` に書くこと。

2. **storage API**（`lib/storage.js`）：`get(key)` → パース済みJSON（無ければ `null`）、`put(key, JSON.stringify(data,null,2))`。既存APIは `loadBlob`/`saveBlob` ラッパーを各ファイルにコピーして使う慣習。

3. **テストの storage スタブ**は `Module._load` 差し替え方式（`test/reserve.test.js` 参照）。R2に触らない。

4. **Windows / Node 25**：`node --test test/`（ディレクトリ指定）は落ちる。**テストファイルを明示**して実行する。

5. **認証**：`req.headers['authorization'] === 'Bearer ' + (process.env.COMINGSOON_PASSWORD||'').trim()`。

6. **税**：内税10%単一。`taxIncluded = round(total - total/1.1)`。

7. **コミット規約**：コミットメッセージ末尾に `Co-Authored-By: Claude Opus 4.8 <noreply@anthropic.com>` を付ける。master 直接ではなく `feat/checkout-inhouse` ブランチで作業する。

/**
 * lib/checkout.js — 会計の純粋計算関数（副作用なし・ストレージ非依存）
 * 小計／割引／内税／お釣り／伝票No採番をここに集約しテスト可能にする。
 */

function calcSubtotal(items) {
  return (items || []).reduce(
    (s, it) => s + (Number(it.price) || 0) * (Number(it.qty) || 0), 0);
}

function applyDiscount(subtotal, discount) {
  let discountAmount = 0;
  if (discount && Number(discount.value) > 0) {
    if (discount.type === 'percent') {
      discountAmount = Math.round(subtotal * Number(discount.value) / 100);
    } else { // 'amount'
      discountAmount = Number(discount.value);
    }
  }
  discountAmount = Math.min(discountAmount, subtotal); // 合計超過を0クランプ
  const total = Math.max(subtotal - discountAmount, 0);
  return { discountAmount, total };
}

function calcTax(total) {
  return Math.round(total - total / 1.1); // 内税10%
}

function calcChange(total, tendered) {
  if (Number(tendered) < total) throw new Error('預かり金額が合計より少ないです');
  return Number(tendered) - total;
}

function nextSlipNo(date, salesSameDay) {
  const ymd = String(date).replace(/-/g, '');
  return `${ymd}-${String((salesSameDay || 0) + 1).padStart(3, '0')}`;
}

function computeCheckout({ items, discount, payment, tendered }) {
  const subtotal = calcSubtotal(items);
  const { discountAmount, total } = applyDiscount(subtotal, discount || { type: 'amount', value: 0 });
  const taxIncluded = calcTax(total);
  const change = payment === 'cash' ? calcChange(total, tendered) : null;
  return { subtotal, discountAmount, total, taxIncluded, change };
}

module.exports = { calcSubtotal, applyDiscount, calcTax, calcChange, nextSlipNo, computeCheckout };
```

`api/products.js`:

/**
 * api/products.js — 物販マスタ CRUD
 *
 * GET    /api/products[?activeOnly=1]   一覧
 * POST   /api/products                  追加({name,price}) or 更新({id,name,price,active})
 * DELETE /api/products?id=p001          論理削除(active:false)
 * 認証: Authorization: Bearer ${COMINGSOON_PASSWORD}
 */

require('dotenv').config();
const storage = require('../lib/storage');

const PRODUCTS_KEY = 'product-master.json';

async function loadBlob() {
  try { return (await storage.get(PRODUCTS_KEY)) || []; } catch { return []; }
}
async function saveBlob(data) {
  await storage.put(PRODUCTS_KEY, JSON.stringify(data, null, 2));
}

function nextId(products) {
  const max = products.reduce((m, p) => {
    const n = Number(String(p.id || '').replace(/^p/, ''));
    return Number.isFinite(n) && n > m ? n : m;
  }, 0);
  return `p${String(max + 1).padStart(3, '0')}`;
}

module.exports = async function handler(req, res) {
  const origin = req.headers.origin || '';
  if (origin) res.setHeader('Access-Control-Allow-Origin', origin);
  res.setHeader('Access-Control-Allow-Methods', 'GET, POST, DELETE, OPTIONS');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization');

  if (req.method === 'OPTIONS') return res.status(200).end();

  const authHeader = req.headers['authorization'] || '';
  if (authHeader !== `Bearer ${(process.env.COMINGSOON_PASSWORD || '').trim()}`) {
    return res.status(401).json({ error: '認証エラー' });
  }

  const products = await loadBlob();

  if (req.method === 'GET') {
    const list = req.query && req.query.activeOnly
      ? products.filter(p => p.active !== false)
      : products;
    return res.status(200).json({ products: list });
  }

  if (req.method === 'POST') {
    const { id, name, price, active } = req.body || {};
    if (!name || typeof price !== 'number') {
      return res.status(400).json({ error: 'name と price(数値) は必須です' });
    }
    if (id) {
      const idx = products.findIndex(p => p.id === id);
      if (idx === -1) return res.status(404).json({ error: '商品が見つかりません' });
      products[idx] = { ...products[idx], name, price, active: active !== false };
      await saveBlob(products);
      return res.status(200).json({ ok: true, product: products[idx] });
    }
    const product = { id: nextId(products), name, price, active: true };
    products.push(product);
    await saveBlob(products);
    return res.status(200).json({ ok: true, product });
  }

  if (req.method === 'DELETE') {
    const id = req.query && req.query.id;
    if (!id) return res.status(400).json({ error: 'id は必須です' });
    const idx = products.findIndex(p => p.id === id);
    if (idx === -1) return res.status(404).json({ error: '商品が見つかりません' });
    products[idx].active = false;
    await saveBlob(products);
    return res.status(200).json({ ok: true });
  }

  return res.status(405).json({ error: 'Method not allowed' });
};
```

async function loadMasters() {
  if (!MENU_MASTER) {

async function openCheckout(reservationId) {
  await loadMasters();
  const rsv = currentReservations.find(r => r.id === reservationId); // 既存のプレート一覧配列名に合わせる
  const items = (rsv.data.menuItems || []).map(m => ({
    kind: 'service', code: m.code || null, name: m.name,
    price: Number(m.price) || 0, qty: Number(m.qty) || 1,
  }));
  coState = { reservationId, customer: rsv.data.name, lines: items };
  document.getElementById('coCustomer').textContent = rsv.data.name + ' 様';
  // メニュー/物販セレクトを埋める
  document.getElementById('coMenuPick').innerHTML =
    '<option value="">＋メニュー追加…</option>' +
    MENU_MASTER.map(m => `<option value="${m.code}">${m.name}（¥${(m.priceIncTax||0).toLocaleString()}）</option>`).join('');
  document.getElementById('coProductPick').innerHTML =
    '<option value="">＋物販追加…</option>' +
    PRODUCT_MASTER.map(p => `<option value="${p.id}">${p.name}（¥${(p.price||0).toLocaleString()}）</option>`).join('');
  document.getElementById('coDiscType').value = 'amount';
  document.getElementById('coDiscValue').value = 0;
  document.getElementById('coTendered').value = '';
  renderCheckout();
  document.getElementById('checkoutModal').style.display = 'flex';
}
function closeCheckout() { document.getElementById('checkoutModal').style.display = 'none'; coState = null; }

function renderCheckout() {
  const tb = document.querySelector('#coItems tbody');
  tb.innerHTML = coState.lines.map((l, i) => `
    <tr>
      <td>${l.kind === 'product' ? '🛍 ' : ''}${l.name}</td>
      <td><input class="co-qty" type="number" min="1" value="${l.qty}" onchange="coSetQty(${i}, this.value)"></td>
      <td>¥${((l.price||0)*(l.qty||1)).toLocaleString()}</td>
      <td><button class="co-del" onclick="coRemove(${i})">×</button></td>
    </tr>`).join('');
  recalcCheckout();
}
function coSetQty(i, v) { coState.lines[i].qty = Math.max(1, Number(v) || 1); renderCheckout(); }
function coRemove(i) { coState.lines.splice(i, 1); renderCheckout(); }

document.addEventListener('change', (e) => {
  if (e.target.id === 'coMenuPick' && e.target.value) {
    const m = MENU_MASTER.find(x => x.code === e.target.value);
    if (m) coState.lines.push({ kind: 'service', code: m.code, name: m.name, price: Number(m.priceIncTax) || 0, qty: 1 });
    e.target.value = ''; renderCheckout();
  }
  if (e.target.id === 'coProductPick' && e.target.value) {
    const p = PRODUCT_MASTER.find(x => x.id === e.target.value);
    if (p) coState.lines.push({ kind: 'product', id: p.id, name: p.name, price: Number(p.price) || 0, qty: 1 });
    e.target.value = ''; renderCheckout();
  }
  if (['coDiscType', 'coDiscValue', 'coTendered'].includes(e.target.id)) recalcCheckout();
  if (e.target.name === 'coPay') recalcCheckout();
});

function recalcCheckout() {
  const subtotal = coState.lines.reduce((s, l) => s + (Number(l.price)||0) * (Number(l.qty)||1), 0);
  const type = document.getElementById('coDiscType').value;
  const value = Number(document.getElementById('coDiscValue').value) || 0;
  let discountAmount = type === 'percent' ? Math.round(subtotal * value / 100) : value;
  discountAmount = Math.min(discountAmount, subtotal);
  const total = Math.max(subtotal - discountAmount, 0);
  const tax = Math.round(total - total / 1.1);
  const pay = document.querySelector('input[name="coPay"]:checked').value;
  document.getElementById('coSubtotal').textContent = '¥' + subtotal.toLocaleString();
  document.getElementById('coTotal').textContent = '¥' + total.toLocaleString();
  document.getElementById('coTax').textContent = '¥' + tax.toLocaleString();
  document.getElementById('coCashRow').style.display = pay === 'cash' ? 'flex' : 'none';
  const tendered = Number(document.getElementById('coTendered').value) || 0;
  const change = pay === 'cash' ? tendered - total : 0;
  document.getElementById('coChange').textContent = '¥' + (change >= 0 ? change : 0).toLocaleString();
  document.getElementById('coConfirmBtn').disabled = pay === 'cash' && tendered < total;
  coState._total = total;
}
